PROGRAM testeproc2;
CONST a=2;
VAR x,y:INTEGER;
PROC EDURE p;
VAR z: INTEGER;
PROCEDURE q;
VAR t: INTEGER;
BEGIN (* inicio da q*)
z:= z - 100 ; t:= z*a;
IF t > 100 THEN CALL q ELSE WRITELN(t)
END; (* fim de q*)
BEGIN (* inicio da P*)
z:= x+y*a; IF z> 100 THEN CALL q ELSE WRITELN(z);
END; (* fim da p*)
BEGIN (* programa principal*)
READLN(x,y);
IF x>1000 THEN x:= 1100
ELSE x::= y+100;
WHILE x>y DO BEGIN CALL p; READLN(x,y) END;
WRITELN(" tudo ok - boas ferias ");
END.
